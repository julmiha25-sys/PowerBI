Задача проекта: создать дашборд, отражающий финансовую деятельность логистической компании (в разрезе временного диапазона, по направлениям бизнеса, видам услуг). Кроме того, оценке подлежит и эффективность менеджеров.
Исходные данные: xls-файл деятельности компании (Исходные данные и Рынок). В полях ГодМесяц таблиц хранится дата - первое число каждого месяца. Т.е. статистика собиралась помесячно и аккумулировалась в файл. Объемы перевозок не зависят от даты (это накопительная информация к моменту анализа при выгрузке).
Изначально в файле имелась связь один ко многим Рынок (Виду слуги Город) и Исходные данные (Вид услуги Филиал). В последствии добавлены связи к таблице Календарь.
Дополнительно создана таблица Календарь для фильтра и Сводные таблицы для ABC-анализа.
 
Сводная = 
SUMMARIZECOLUMNS(
    'Исходные данные'[Вид услуги],
    'Исходные данные'[ГодМесяц],
    "Общая факт прибыль", SUM('Исходные данные'[Факт прибыль])
)

Сводная по направлениям = 
SUMMARIZECOLUMNS(
    'Исходные данные'[Направление бизнеса],
    'Исходные данные'[ГодМесяц],
    "Общая факт прибыль", SUM('Исходные данные'[Факт прибыль])
)

Используемые меры:
% от плана = 
VAR plan_profit=SUM('Исходные данные'[План прибыль])
VAR fact_profit=SUM('Исходные данные'[Факт прибыль])
RETURN
    IF(
        NOT ISBLANK(fact_profit),
        DIVIDE(fact_profit-plan_profit,plan_profit)
    )

% от плана выручка = 
VAR plan_revenue=SUM('Исходные данные'[План выручка])
VAR fact_revenue=SUM('Исходные данные'[Факт выручка])
RETURN
    IF(
        NOT ISBLANK(fact_revenue),
        DIVIDE(fact_revenue-plan_revenue, plan_revenue)
    )

Доля рынка = SUM('Исходные данные'[Факт выручка])/SUM('Рынок'[Объем перевозок])
Кол-во менеджеров = 
CALCULATE(
    DISTINCTCOUNT('Исходные данные'[Менежер]),
    ALL('Исходные данные'[Менежер]) /* Убираем все фильтры кроме фильтра по менеджеру*/
)

Маржинальность = 
DIVIDE(
    SUM('Исходные данные'[Факт прибыль]),
    SUM('Исходные данные'[Факт выручка]),
    0
)

Ранг менеджера по марж = 
IF(
    HASONEVALUE('Исходные данные'[Менежер]),  /* Если менеджер на фильтре выбран */
    RANKX(
        ALL('Исходные данные'[Менежер]),
        ROUND([Маржинальность], 5),,DESC
    ), 0  /* Если менеджер не выбран, то на датчике пусто */
)

Рост к пред кварталу = 
VAR Current_quarter = SUM('Исходные данные'[Факт прибыль])
VAR Last_quarter = 
    CALCULATE(
        SUM('Исходные данные'[Факт прибыль]),
        DATEADD('Календарь'[Date], -1, QUARTER) /* Сдвигаем дату на месяц назад */
    )
RETURN
IF(
    ISBLANK(Last_quarter) || Last_quarter = 0,
    BLANK(),  /* Не отражаем меру, если нет данных за прошлый квартал*/
    DIVIDE(Current_quarter - Last_quarter, Last_quarter)
)




Средний чек = 
AVERAGEX(
    VALUES('Исходные данные'[Клиент]),
    CALCULATE(SUM('Исходные данные'[Факт выручка]))
)

ABC = /* Расчет ABC для видов услуг */
VAR Current_profit = /* Текущее значение прибыли */
    SUM('Сводная'[Общая факт прибыль]) /* Убираем фильтр по услугам с круговой диаграммы */
VAR All_profit = /* Временная таблица со всеми видами услуг и их прибылью */
    ADDCOLUMNS(
        ALLSELECTED('Сводная'[Вид услуги]),
        "Прибыль по услугам", CALCULATE(SUM('Сводная'[Общая факт прибыль]))
    )
VAR Cummulative_sum_profit = SUMX(FILTER(All_profit, [Прибыль по услугам] >= Current_profit), [Прибыль по услугам]) /* Накопленная сумма прибыли всех услуг, у которых прибыль больше или равна текущей*/
VAR Total_profit = SUMX(All_profit, [Прибыль по услугам])  /* Общая сумма прибыли всех услуг в выборке */
VAR Share = DIVIDE(Cummulative_sum_profit, Total_profit) /* Накопленная доля текущей услуги и всех вышестоящих */
RETURN
IF(Share <= 0.7, "A", IF(Share <= 0.9, "B", "C")) 

ABC_напр = /* Расчет ABC для направления бизнеса */
VAR Current_profit = /* Текущее значение прибыли */
    SUM('Сводная по направлениям'[Общая факт прибыль]) 
VAR All_profit = /* Временная таблица со всеми направлениями и их прибылью */
    ADDCOLUMNS(
        ALLSELECTED('Сводная по направлениям'[Направление бизнеса]),
        "Прибыль по направлениям", CALCULATE(SUM('Сводная по направлениям'[Общая факт прибыль]))
    )
VAR Cummulative_sum_profit = SUMX(FILTER(All_profit, [Прибыль по направлениям] >= Current_profit), [Прибыль по направлениям]) /* Накопленная сумма прибыли всех направлений, у которых прибыль больше или равна текущей*/
VAR Total_profit = SUMX(All_profit, [Прибыль по направлениям])  /* Общая сумма прибыли всех направлений в выборке */
VAR Share = DIVIDE(Cummulative_sum_profit, Total_profit) /* Накопленная доля текущего направления и всех вышестоящих */
RETURN
IF(Share <= 0.7, "A", IF(Share <= 0.9, "B", "C"))
